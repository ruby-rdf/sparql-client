def to_s
  query_text = 'LOAD '
  query_text += 'SILENT ' if self.options[:silent]
  query_text += SPARQL::Client.serialize_uri(@from)
  query_text += ' INTO GRAPH ' + SPARQL::Client.serialize_uri(@into) if @into
  query_text
end
